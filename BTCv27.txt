""" Transaction model, UTXO set management, and script execution. Design decisions: - Outputs use Pedersen commitments for value hiding (privacy). - Each transaction carries an optional zk-SNARK proof for validity without revealing input/output values. - HTLC support for atomic swaps and payment channels is first-class. - Signatures use Dilithium3 (NIST PQC Level 3) — imported lazily to avoid circular dependency with crypto package. """ from __future__ import annotations import hashlib import time import json from dataclasses import dataclass, field from typing import Any, Dict, List, Optional import msgpack import zlib from btcv26.core.exceptions import ValidationError @dataclass class TransactionInput: """Reference to a previous transaction output (UTXO being spent).""" tx_id: str output_index: int # Signature and public key are set after construction signature: Optional[str] = None public_key: Optional[str] = None def to_dict(self) -> Dict: return { "tx_id": self.tx_id, "output_index": self.output_index, "signature": self.signature, "public_key": self.public_key, } @classmethod def from_dict(cls, d: Dict) -> "TransactionInput": return cls(**d) @dataclass class TransactionOutput: """ A transaction output with a Pedersen commitment for value confidentiality. """ recipient: str commitment: str # Pedersen commitment hex value: float # Plaintext value — only visible to sender/recipient hash_lock: Optional[str] = None # HTLC: SHA256 of preimage time_lock: Optional[int] = None # HTLC: expiry block height def to_dict(self) -> Dict: return { "recipient": self.recipient, "commitment": self.commitment, "value": self.value, "hash_lock": self.hash_lock, "time_lock": self.time_lock, } @classmethod def from_dict(cls, d: Dict) -> "TransactionOutput": return cls(**d) @dataclass class Script: """ Minimal scripting for smart contract conditions. Returns True if no proof is attached (proof is optional for transparent transactions). """ if self.zk_proof is None: return True try: from btcv26.crypto.zkp import ZKProofSystem zkp = ZKProofSystem() return zkp.verify_balance_proof( proof_hex=self.zk_proof, inputs=[i.to_dict() for i in self.inputs], outputs=[o.to_dict() for o in self.outputs], ) except Exception: return False def verify_htlc(self) -> bool: """ Verify HTLC conditions for outputs that carry a hash_lock. For a hash_locked output the transaction must include the preimage in script.params['preimage'] whose SHA-256 matches hash_lock. """ for output in self.outputs: if output.hash_lock is not None: preimage = self.script.params.get("preimage") if preimage is None: return False if hashlib.sha256(preimage.encode()).hexdigest() != output.hash_lock: return False return True async def execute_script(self, storage) -> bool: """ Execute the transaction script against current chain state. Returns True if the script passes, False otherwise. """ stype = self.script.type params = self.script.params if stype == "transfer": return True if stype == "balance_check": address = params.get("address") min_balance = params.get("min_balance", 0.0) balance = await storage.get_balance(address) return balance >= min_balance if stype == "multisig": # Validate M-of-N: params['signatures'] list, params['public_keys'] list from btcv26.crypto.pqc import PQCrypto pqc = PQCrypto() required = params.get("required", 1) signatures = params.get("signatures", []) public_keys = params.get("public_keys", []) message = f"{self.chain_id}{self.block_height}{self.tx_id}".encode() valid_count = sum( 1 for sig, pk in zip(signatures, public_keys) if pqc.verify_dilithium(message, sig, pk) ) return valid_count >= required if stype == "htlc": return self.verify_htlc() raise ValidationError(f"Unknown script type: {stype!r}") def is_coinbase(self) -> bool: """Return True if this is a coinbase (mining reward) transaction.""" return ( len(self.inputs) == 1 and self.inputs[0].tx_id == "coinbase" and self.inputs[0].output_index == -1 ) def __repr__(self) -> str: return f"Transaction(tx_id={self.tx_id[:12]}..., inputs={len(self.inputs)}, outputs={len(self.outputs)})"

""" Transaction mempool with priority queue, double-spend detection, and configurable validation pipeline. """ from __future__ import annotations import asyncio import logging import time from collections import defaultdict from typing import Callable, Dict, List, Optional, Set from btcv26.core.exceptions import ValidationError from btcv26.core.transaction import Transaction logger = logging.getLogger(__name__) class Mempool: """ In-memory transaction pool. Transactions are sorted by fee-per-byte descending so miners select the most profitable bundle. Double-spends are detected by tracking which UTXOs are claimed by pending transactions. Parameters ---------- max_size : int Maximum number of pending transactions before eviction of lowest-fee txs. max_tx_age : float Seconds after which a transaction is considered stale and removed. """ def __init__(self, max_size: int = 10_000, max_tx_age: float = 3600.0): self.transactions: Dict[str, Transaction] = {} self._spent_utxos: Dict[str, str] = {} # "txid:idx" -> claiming tx_id self._fee_index: Dict[str, float] = {} # tx_id -> fee self.max_size = max_size self.max_tx_age = max_tx_age self._lock = asyncio.Lock() async def add( self, tx: Transaction, validator: Optional[Callable] = None, ) -> None: """ Add a transaction to the mempool. Parameters ---------- tx : The transaction to add. validator : Optional async or sync callable(tx) -> bool. If provided, the transaction is rejected when it returns False. """ async with self._lock: if tx.tx_id in self.transactions: return # already present # Run validator if validator is not None: try: result = validator(tx) if asyncio.iscoroutine(result): result = await result if not result: raise ValidationError(f"Transaction {tx.tx_id} failed validation") except ValidationError: raise except Exception as exc: raise ValidationError(f"Validator raised exception: {exc}") from exc # Check for double spends for inp in tx.inputs: utxo_key = f"{inp.tx_id}:{inp.output_index}" if utxo_key in self._spent_utxos: existing = self._spent_utxos[utxo_key] raise ValidationError( f"Double spend detected: UTXO {utxo_key} already claimed by {existing}" ) # Evict lowest-fee tx if at capacity if len(self.transactions) >= self.max_size: self._evict_lowest_fee() # Register for inp in tx.inputs: utxo_key = f"{inp.tx_id}:{inp.output_index}" self._spent_utxos[utxo_key] = tx.tx_id self.transactions[tx.tx_id] = tx self._fee_index[tx.tx_id] = tx.fee logger.debug("Mempool: added %s (fee=%.8f)", tx.tx_id[:12], tx.fee) def remove(self, tx_id: str) -> None: """Remove a transaction (e.g., after it is included in a block).""" tx = self.transactions.pop(tx_id, None) if tx: for inp in tx.inputs: self._spent_utxos.pop(f"{inp.tx_id}:{inp.output_index}", None) self._fee_index.pop(tx_id, None) def get_for_block(self, max_txs: int = 500) -> List[Transaction]: """Return up to max_txs transactions sorted by descending fee.""" sorted_ids = sorted( self._fee_index, key=lambda tid: self._fee_index[tid], reverse=True ) return [self.transactions[tid] for tid in sorted_ids[:max_txs]] def prune_stale(self) -> int: """Remove transactions older than max_tx_age. Returns count removed.""" cutoff = time.time() - self.max_tx_age stale = [ tid for tid, tx in self.transactions.items() if tx.timestamp < cutoff ] for tid in stale: self.remove(tid) return len(stale) def _evict_lowest_fee(self) -> None: if not self._fee_index: return lowest = min(self._fee_index, key=lambda tid: self._fee_index[tid]) self.remove(lowest) logger.debug("Mempool: evicted low-fee tx %s", lowest[:12]) def __len__(self) -> int: return len(self.transactions) 

btcv26/core/chain.py

""" Chain state: canonical chain, UTXO set, cumulative work tracking. The Chain class is intentionally thin — it owns the list of accepted blocks and the UTXO set derived from them. Consensus rules (PoW validation, fork choice) live in btcv26.consensus. Storage I/O lives in btcv26.storage. """ from __future__ import annotations import asyncio import logging from typing import Dict, List, Optional, Tuple from btcv26.core.block import Block from btcv26.core.exceptions import BlockchainError, ValidationError from btcv26.core.transaction import Transaction logger = logging.getLogger(__name__) # Work contributed by a block of a given difficulty level. # Mirrors Bitcoin: work ≈ 2^difficulty def _block_work(difficulty: int) -> int: return 2 ** difficulty class UTXOSet: """ In-memory UTXO set. Keys are "tx_id:output_index" strings. Values are TransactionOutput dicts. For a production deployment this must be backed by the sharded LevelDB storage; the in-memory version here is used by tests and the genesis block. """ def __init__(self): self._utxos: Dict[str, dict] = {} def apply_block(self, block: Block) -> None: """Update the UTXO set by applying all transactions in a block.""" for tx in block.transactions: if not tx.is_coinbase(): for inp in tx.inputs: key = f"{inp.tx_id}:{inp.output_index}" self._utxos.pop(key, None) for idx, out in enumerate(tx.outputs): key = f"{tx.tx_id}:{idx}" self._utxos[key] = out.to_dict() def rollback_block(self, block: Block) -> None: """ Undo a block (used during chain reorganisation). NOTE: Rolling back inputs requires the previous outputs — callers must restore spent UTXOs from storage before calling this. """ for tx in block.transactions: for idx in range(len(tx.outputs)): key = f"{tx.tx_id}:{idx}" self._utxos.pop(key, None) def exists(self, tx_id: str, output_index: int) -> bool: return f"{tx_id}:{output_index}" in self._utxos def get(self, tx_id: str, output_index: int) -> Optional[dict]: return self._utxos.get(f"{tx_id}:{output_index}") def get_balance(self, address: str) -> float: return sum( o["value"] for o in self._utxos.values() if o.get("recipient") == address ) class Chain: """ Manages the canonical blockchain and orphan tracking. Parameters ---------- genesis : Block The genesis block. Created by the caller so tests can inject deterministic genesis blocks. """ def __init__(self, genesis: Block): self.chain: List[Block] = [genesis] self.utxo_set = UTXOSet() self.utxo_set.apply_block(genesis) self._cumulative_work: Dict[str, int] = { genesis.calculate_hash(): _block_work(genesis.difficulty) } self._orphans: Dict[str, Block] = {} # hash -> block self._lock = asyncio.Lock() # ------------------------------------------------------------------ # Read helpers # ------------------------------------------------------------------ def tip(self) -> Block: """Return the most recent block on the canonical chain.""" return self.chain[-1] def height(self) -> int: return len(self.chain) - 1 def get_by_index(self, index: int) -> Optional[Block]: if 0 <= index < len(self.chain): return self.chain[index] return None def get_by_hash(self, block_hash: str) -> Optional[Block]: for block in reversed(self.chain): if block.calculate_hash() == block_hash: return block return None def cumulative_work(self, block_hash: Optional[str] = None) -> int: if block_hash is None: block_hash = self.tip().calculate_hash() return self._cumulative_work.get(block_hash, 0) # ------------------------------------------------------------------ # Append # ------------------------------------------------------------------ async def append(self, block: Block) -> None: """ Append a validated block to the canonical chain. Does NOT re-validate — callers (consensus.validator) must validate before calling this. """ async with self._lock: self.chain.append(block) h = block.calculate_hash() parent_work = self._cumulative_work.get(block.previous_hash, 0) self._cumulative_work[h] = parent_work + _block_work(block.difficulty) self.utxo_set.apply_block(block) # ------------------------------------------------------------------ # Fork resolution # ------------------------------------------------------------------ async def resolve_fork( self, candidate: Block, candidate_chain: List[Block], ) -> bool: """ Compare cumulative work of the candidate chain against the canonical chain. If the candidate has more work, reorganise to it. Returns True if a reorganisation occurred. """ async with self._lock: candidate_work = sum( _block_work(b.difficulty) for b in candidate_chain ) current_work = self.cumulative_work() if candidate_work <= current_work: logger.debug( "Fork rejected: candidate_work=%d <= current_work=%d", candidate_work, current_work, ) return False # Find common ancestor common_ancestor_idx = self._find_common_ancestor(candidate_chain) if common_ancestor_idx is None: logger.warning("Fork: no common ancestor found, ignoring") return False # Roll back canonical chain to common ancestor rolled_back = self.chain[common_ancestor_idx + 1 :] for block in reversed(rolled_back): self.utxo_set.rollback_block(block) self.chain = self.chain[: common_ancestor_idx + 1] # Apply the new fork for block in candidate_chain[common_ancestor_idx + 1 :]: self.chain.append(block) h = block.calculate_hash() parent_work = self._cumulative_work.get(block.previous_hash, 0) self._cumulative_work[h] = parent_work + _block_work(block.difficulty) self.utxo_set.apply_block(block) logger.info( "Chain reorganised to height %d (rolled back %d blocks)", self.height(), len(rolled_back), ) return True def _find_common_ancestor(self, candidate_chain: List[Block]) -> Optional[int]: candidate_hashes = {b.calculate_hash() for b in candidate_chain} for idx, block in enumerate(self.chain): if block.calculate_hash() in candidate_hashes: return idx return None

btcv26/crypto/pqc.py

""" Post-quantum cryptography using NIST-standardised algorithms via liboqs. Algorithms ---------- - Dilithium3 (ML-DSA-65) : Digital signatures — replaces ECDSA/Schnorr. - Kyber768 (ML-KEM-768) : Key encapsulation — replaces ECDH for session keys. Dependency: `pip install pyoqs` (Python bindings for Open Quantum Safe liboqs) If pyoqs is unavailable, the module falls back to a DEVELOPMENT-ONLY stub that uses Ed25519 via PyNaCl. This stub MUST NOT be used in production. """ from __future__ import annotations import base64 import hashlib import logging import os from typing import Tuple logger = logging.getLogger(__name__) # --------------------------------------------------------------------------- # Try real pyoqs first, then fall back to a test stub # --------------------------------------------------------------------------- try: import oqs # type: ignore _OQS_AVAILABLE = True logger.info("pyoqs available — using Dilithium3 and Kyber768") except ImportError: _OQS_AVAILABLE = False logger.warning( "pyoqs not installed — using Ed25519 stub (DEVELOPMENT ONLY, NOT PRODUCTION)" ) class PQCrypto: """ Unified interface for post-quantum signing and key encapsulation. """ Per-peer token-bucket rate limiting and DoS detection. Each peer gets a token bucket refilled at RATE tokens/second. A request costs 1 token. When the bucket is empty the peer is throttled. """ from __future__ import annotations import asyncio import logging import time from collections import defaultdict from typing import Dict, Set from prometheus_client import Counter # type: ignore logger = logging.getLogger(__name__) dos_attempts = Counter("dos_attempts_total", "Total detected DoS attempts") class TokenBucket: def __init__(self, rate: float, capacity: float): self.rate = rate self.capacity = capacity self._tokens = capacity self._last_refill = time.monotonic() def consume(self, tokens: float = 1.0) -> bool: now = time.monotonic() elapsed = now - self._last_refill self._tokens = min(self.capacity, self._tokens + elapsed * self.rate) self._last_refill = now if self._tokens >= tokens: self._tokens -= tokens return True return False class RateLimiter: """ Per-peer rate limiter with ban list. Parameters ---------- rate : Tokens per second per peer. capacity : Burst capacity per peer. ban_threshold : How many rate-limit violations before banning. ban_duration : How long to ban a peer (seconds). """ def __init__( self, rate: float = 100.0, capacity: float = 200.0, ban_threshold: int = 10, ban_duration: float = 300.0, ): self._buckets: Dict[str, TokenBucket] = {} self._violations: Dict[str, int] = defaultdict(int) self._banned_until: Dict[str, float] = {} self.rate = rate self.capacity = capacity self.ban_threshold = ban_threshold self.ban_duration = ban_duration def check(self, peer_id: str) -> bool: """ Return True if the request is allowed, False if throttled/banned. """ now = time.monotonic() # Check ban ban_until = self._banned_until.get(peer_id, 0) if now < ban_until: return False # Get or create bucket if peer_id not in self._buckets: self._buckets[peer_id] = TokenBucket(self.rate, self.capacity) if self._buckets[peer_id].consume(): return True # Rate limit hit self._violations[peer_id] += 1 dos_attempts.inc() logger.warning("Rate limit exceeded for peer %s (violations=%d)", peer_id, self._violations[peer_id]) if self._violations[peer_id] >= self.ban_threshold: self._banned_until[peer_id] = now + self.ban_duration logger.warning("Peer %s banned for %.0fs", peer_id, self.ban_duration) return False def is_banned(self, peer_id: str) -> bool: return time.monotonic() < self._banned_until.get(peer_id, 0)

""" Payment channel state machine (Lightning Network-style). A payment channel is opened by locking funds in a 2-of-2 multisig output. Payments are made by exchanging signed state updates off-chain. """ def __init__(self, initiator: str, responder: str, capacity: float): self.initiator = initiator self.responder = responder self.capacity = capacity self.state = ChannelState.OPEN self.updates: List[ChannelUpdate] = [] self.opened_at = time.time() self.closed_at: Optional[float] = None # Initial state: all funds with initiator initial = ChannelUpdate( sequence=0, initiator_balance=capacity, responder_balance=0.0, ) self.updates.append(initial) @property def latest_update(self) -> ChannelUpdate: return self.updates[-1] def apply_update(self, update: ChannelUpdate) -> None: """ Apply a new signed state update. """ Simplified Payment Verification (SPV) proof verification. Used by the two-way peg to verify that a Bitcoin peg-in transaction was confirmed without trusting the RPC node. An SPV proof consists of: - The transaction being proved. - The Merkle branch from the transaction to the block Merkle root. - The block header (for PoW and timestamp verification). This module verifies the Merkle branch against the block header's merkle_root field using SHA-256d (double-SHA256) as per Bitcoin. """ from __future__ import annotations import hashlib from dataclasses import dataclass from typing import Dict, List def _sha256d(data: bytes) -> bytes: """Bitcoin's double-SHA256.""" return hashlib.sha256(hashlib.sha256(data).digest()).digest() def _txid_to_bytes(txid_hex: str) -> bytes: """Convert a txid hex string to little-endian bytes (as Bitcoin stores it).""" return bytes.fromhex(txid_hex)[::-1] @dataclass class SPVProof: """ Bitcoin SPV inclusion proof. Attributes ---------- txid : The Bitcoin transaction ID being proved (hex, big-endian). merkle_branch : List of sibling hashes on the path to the Merkle root. Each entry: {'hash': str, 'position': 'left'|'right'}. block_header : The block header dict (from Bitcoin RPC). tx_index : The transaction's index within the block. confirmations : Number of confirmations at proof generation time. """ txid: str merkle_branch: List[Dict] block_header: Dict tx_index: int confirmations: int class SPVVerifier: """Verifies Bitcoin SPV proofs.""" @staticmethod def verify(proof: SPVProof, min_confirmations: int = 6) -> bool: """ Verify an SPV proof. Returns True iff: 1. The merkle branch connects txid to block_header.merkleroot. 2. The block has >= min_confirmations. """ if proof.confirmations < min_confirmations: return False # Reconstruct the Merkle root from the branch current = _txid_to_bytes(proof.txid) for step in proof.merkle_branch: sibling = bytes.fromhex(step["hash"])[::-1] if step["position"] == "right": combined = current + sibling else: combined = sibling + current current = _sha256d(combined) # The reconstructed root should match the block header expected_root = bytes.fromhex(proof.block_header.get("merkleroot", ""))[::-1] return current == expected_root

""" Two-way peg between Bitcoin and BTCv26. Security model -------------- - PEG-IN : The user locks BTC in a federation multisig address on Bitcoin. The federation releases the corresponding BTC to the user's Bitcoin address. Federation ---------- The federation is a 3-of-5 multisig (configurable). Federation members each hold one Dilithium3 key for signing BTCv26 transactions and one Bitcoin key for the multisig address. CRITICAL SECURITY NOTES: - The peg-in flow validates the Bitcoin transaction via SPV proof. Never trust user-supplied amounts without on-chain verification. - Peg-out requires burning BTCV before releasing BTC (atomicity). - Replay attacks are prevented by recording processed Bitcoin txids. - Federation key ceremony and HSM integration are out of scope here. """ from __future__ import annotations import hashlib import logging import time from typing import Dict, Optional, Set from btcv26.core.exceptions import PegError from btcv26.core.transaction import Transaction, TransactionInput, TransactionOutput from btcv26.crosschain.bitcoin_rpc import BitcoinRPCClient from btcv26.crosschain.spv import SPVProof, SPVVerifier logger = logging.getLogger(__name__) # Minimum confirmations before minting BTCV MIN_CONFIRMATIONS = 6 # Exchange rate: 1 BTC = 1 BTCV (1:1 peg) PEG_RATIO = 1.0 class TwoWayPeg: """ Two-way peg manager. Parameters ---------- rpc : Bitcoin RPC client. fed_address : The federation's Bitcoin multisig deposit address. """ def __init__(self, rpc: BitcoinRPCClient, fed_address: str): self.rpc = rpc self.fed_address = fed_address self._processed_btc_txids: Set[str] = set() # Replay protection async def create_peg_in( self, btc_txid: str, recipient: str, spv_proof: Optional[SPVProof] = None, ) -> Transaction: """ Verify a Bitcoin peg-in and mint equivalent BTCV. Parameters ---------- btc_txid : The Bitcoin transaction that sent BTC to the federation. recipient : BTCv26 address to receive minted BTCV. spv_proof : SPV proof of the Bitcoin transaction (required in production). Returns ------- A BTCv26 Transaction minting the equivalent BTCV amount. """ # Replay protection if btc_txid in self._processed_btc_txids: raise PegError(f"Bitcoin txid {btc_txid} already processed") # Verify the Bitcoin transaction via RPC try: btc_tx = await self.rpc.get_raw_transaction(btc_txid) except Exception as exc: raise PegError(f"Cannot retrieve Bitcoin tx {btc_txid}: {exc}") from exc # Check confirmations confirmations = btc_tx.get("confirmations", 0) if confirmations < MIN_CONFIRMATIONS: raise PegError( f"Bitcoin tx {btc_txid} has only {confirmations} confirmations " f"(need {MIN_CONFIRMATIONS})" ) # Verify SPV proof if provided if spv_proof is not None: if not SPVVerifier.verify(spv_proof, min_confirmations=MIN_CONFIRMATIONS): raise PegError(f"SPV proof invalid for Bitcoin tx {btc_txid}") # Extract the amount sent to the federation address amount = self._extract_federation_output(btc_tx) if amount <= 0: raise PegError(f"No output to federation address in Bitcoin tx {btc_txid}") # Mark as processed self._processed_btc_txids.add(btc_txid) # Create the mint transaction mint_amount = amount * PEG_RATIO commitment = hashlib.sha256(f"{recipient}:{mint_amount}".encode()).hexdigest() tx = Transaction( inputs=[{"tx_id": "coinbase", "output_index": -1}], outputs=[{ "recipient": recipient, "commitment": commitment, "value": mint_amount, }], fee=0.0, nonce=0, script={"type": "transfer", "params": {"peg_in": btc_txid}}, ) logger.info( "Peg-in: minted %.8f BTCV to %s (BTC txid=%s)", mint_amount, recipient, btc_txid, ) return tx async def create_peg_out( self, burn_tx: Transaction, btc_destination: str, ) -> str: """ Process a BTCV burn and release BTC to the user's Bitcoin address. Parameters ---------- burn_tx : The BTCv26 transaction burning BTCV. btc_destination : The user's Bitcoin address to receive BTC. Returns ------- The Bitcoin transaction ID of the release. """ # Sum the outputs being burned amount = sum(o.value for o in burn_tx.outputs) if amount <= 0: raise PegError("Peg-out amount must be positive") # In production: build and broadcast a real Bitcoin transaction # For now, log and return a deterministic mock txid mock_btc_txid = hashlib.sha256( f"pegout:{burn_tx.tx_id}:{btc_destination}:{amount}".encode() ).hexdigest() logger.info( "Peg-out: releasing %.8f BTC to %s (BTCV txid=%s, BTC txid=%s)", amount / PEG_RATIO, btc_destination, burn_tx.tx_id, mock_btc_txid, ) return mock_btc_txid def _extract_federation_output(self, btc_tx: Dict) -> float: """Sum all outputs in a Bitcoin transaction that go to the federation address.""" total = 0.0 for vout in btc_tx.get("vout", []): addresses = ( vout.get("scriptPubKey", {}).get("addresses", []) or [vout.get("scriptPubKey", {}).get("address", "")] ) if self.fed_address in addresses: total += float(vout.get("value", 0)) return total

""" proposal = self._proposals.get(proposal_id) if proposal is None: raise GovernanceError(f"Proposal {proposal_id} not found") if proposal.status != ProposalStatus.ACTIVE: raise GovernanceError( f"Proposal {proposal_id} is not active (status={proposal.status.value})" ) if proposal.is_expired(): proposal.status = ProposalStatus.EXPIRED raise GovernanceError(f"Proposal {proposal_id} has expired") if voter in proposal.votes: raise GovernanceError(f"Voter {voter} has already voted on {proposal_id}") proposal.votes[voter] = Vote( voter=voter, vote=vote, weight=weight, signature=signature, ) logger.info( "Vote recorded: proposal=%s voter=%s vote=%s weight=%.2f", proposal_id[:12], voter, "yes" if vote else "no", weight, ) self._check_outcome(proposal) def _check_outcome(self, proposal: Proposal) -> None: """Auto-approve or reject if thresholds are reached.""" quorum_met = proposal.total_weight >= self.total_supply * QUORUM if not quorum_met: return ratio = proposal.approval_ratio() if ratio > APPROVAL_THRESHOLD: proposal.status = ProposalStatus.APPROVED logger.info( "Proposal %s APPROVED (ratio=%.1f%%)", proposal.proposal_id[:12], ratio * 100, ) elif ratio < (1 - APPROVAL_THRESHOLD): proposal.status = ProposalStatus.REJECTED logger.info("Proposal %s REJECTED", proposal.proposal_id[:12]) def get_proposal(self, proposal_id: str) -> Optional[Proposal]: return self._proposals.get(proposal_id) def list_proposals(self, status: Optional[ProposalStatus] = None) -> List[Proposal]: proposals = list(self._proposals.values()) if status: proposals = [p for p in proposals if p.status == status] return proposals
