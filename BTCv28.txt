""" Transaction model, UTXO set management, and script execution. Each input must carry a valid Dilithium3 signature over: blake3(chain_id || block_height || tx_id) """ try: from btcv27.crypto.pqc import PQCrypto pqc = PQCrypto() message = f"{self.chain_id}{self.block_height}{self.tx_id}".encode() for inp in self.inputs: if inp.signature is None or inp.public_key is None: return False if not pqc.verify_dilithium(message, inp.signature, inp.public_key): return False return True except Exception: return False def verify_zk_proof(self) -> bool: """ Verify the Groth16 proof for confidential value balance. Returns True if no proof is attached (proof is optional for transparent txs). """ if self.zk_proof is None: return True try: from btcv27.crypto.zkp import ZKProofSystem zkp = ZKProofSystem() return zkp.verify_balance_proof( proof_hex=self.zk_proof, inputs=[i.to_dict() for i in self.inputs], outputs=[o.to_dict() for o in self.outputs], ) except Exception: return False def verify_htlc(self) -> bool: """ Verify HTLC conditions for outputs that carry a hash_lock. The preimage must be present in script.params['preimage']. """ for output in self.outputs: if output.hash_lock is not None: preimage = self.script.params.get("preimage") if preimage is None: return False if hashlib.sha256(preimage.encode()).hexdigest() != output.hash_lock: return False return True def verify_channel_close(self, current_height: int) -> bool: """ Verify a channel_close script. Required params: - party_a_sig : Dilithium3 sig from party A over the channel state - party_b_sig : Dilithium3 sig from party B over the channel state - party_a_pk : Party A public key - party_b_pk : Party B public key - channel_state : Dict with {channel_id, seq_num, balance_a, balance_b} - close_type : 'cooperative' | 'unilateral' - dispute_height : For unilateral, the block height after which funds unlock For cooperative close: both sigs required, immediate settlement. For unilateral close: one sig + dispute_height timelock enforced. """ try: from btcv27.crypto.pqc import PQCrypto pqc = PQCrypto() params = self.script.params channel_state = params.get("channel_state", {}) state_bytes = json.dumps(channel_state, sort_keys=True).encode() close_type = params.get("close_type", "cooperative") if close_type == "cooperative": # Both parties must sign the final state for sig_key, pk_key in [ ("party_a_sig", "party_a_pk"), ("party_b_sig", "party_b_pk"), ]: sig = params.get(sig_key) pk = params.get(pk_key) if not sig or not pk: return False if not pqc.verify_dilithium(state_bytes, sig, pk): return False return True elif close_type == "unilateral": # One party initiates; dispute window must have elapsed dispute_height = params.get("dispute_height", 0) if current_height < dispute_height: return False # Still in dispute window # Verify initiating party's signature for sig_key, pk_key in [ ("party_a_sig", "party_a_pk"), ("party_b_sig", "party_b_pk"), ]: sig = params.get(sig_key) pk = params.get(pk_key) if sig and pk: return pqc.verify_dilithium(state_bytes, sig, pk) return False return False except Exception: return False async def execute_script(self, storage, current_height: int = 0) -> bool: """ Execute the transaction script against current chain state. Returns True if the script passes, False otherwise. """ stype = self.script.type params = self.script.params if stype == "transfer": return True if stype == "balance_check": address = params.get("address") min_balance = params.get("min_balance", 0.0) balance = await storage.get_balance(address) return balance >= min_balance if stype == "multisig": from btcv27.crypto.pqc import PQCrypto pqc = PQCrypto() required = params.get("required", 1) signatures = params.get("signatures", []) public_keys = params.get("public_keys", []) message = f"{self.chain_id}{self.block_height}{self.tx_id}".encode() valid_count = sum( 1 for sig, pk in zip(signatures, public_keys) if pqc.verify_dilithium(message, sig, pk) ) return valid_count >= required if stype == "htlc": return self.verify_htlc() if stype == "channel_close": return self.verify_channel_close(current_height) raise ValidationError(f"Unknown script type: {stype!r}") def is_coinbase(self) -> bool: """Return True if this is a coinbase (mining reward) transaction.""" return ( len(self.inputs) == 1 and self.inputs[0].tx_id == "coinbase" and self.inputs[0].output_index == -1 ) def __repr__(self) -> str: return ( f"Transaction(tx_id={self.tx_id[:12]}..., " f"inputs={len(self.inputs)}, outputs={len(self.outputs)})" )

""" Transaction mempool with double-spend detection and fee-priority ordering. """ from __future__ import annotations import asyncio import logging from typing import Callable, Dict, List, Optional from btcv27.core.transaction import Transaction from btcv27.core.exceptions import ValidationError logger = logging.getLogger(__name__) class Mempool: """ In-memory transaction pool. Maintains a spent_utxos set for O(1) double-spend detection. Returns transactions sorted by fee (descending) for block packing. """ def __init__(self): self.transactions: Dict[str, Transaction] = {} self._spent_utxos: set = set() async def add( self, tx: Transaction, validator: Optional[Callable[[Transaction], bool]] = None, ) -> None: if tx.tx_id in self.transactions: return # Idempotent if validator is not None and not validator(tx): raise ValidationError("Transaction rejected by validator.") # Double-spend check for inp in tx.inputs: key = (inp.tx_id, inp.output_index) if key in self._spent_utxos: raise ValidationError( f"Double spend detected: {inp.tx_id}:{inp.output_index}" ) # Register spent UTXOs for inp in tx.inputs: self._spent_utxos.add((inp.tx_id, inp.output_index)) self.transactions[tx.tx_id] = tx logger.debug("Mempool added tx %s (fee=%.6f)", tx.tx_id[:12], tx.fee) def remove(self, tx_id: str) -> None: tx = self.transactions.pop(tx_id, None) if tx: for inp in tx.inputs: self._spent_utxos.discard((inp.tx_id, inp.output_index)) def get_for_block(self, max_txs: int = 500) -> List[Transaction]: """Return transactions sorted by fee descending, capped at max_txs.""" return sorted( self.transactions.values(), key=lambda tx: tx.fee, reverse=True, )[:max_txs] def __len__(self) -> int: return len(self.transactions)

Returns the mined block, or None if max_nonce exceeded. """ for nonce in range(max_nonce): block.nonce = nonce if block.meets_difficulty(): logger.debug( "Block %d mined with nonce %d (hash=%s...)", block.index, nonce, block.calculate_hash()[:12] ) return block logger.warning("Mining failed to find nonce within %d attempts.", max_nonce) return None def adjust_difficulty(chain, interval: int = DIFFICULTY_ADJUSTMENT_INTERVAL) -> int: """ Retarget difficulty every `interval` blocks to maintain TARGET_BLOCK_TIME. """ height = chain.height() if height < interval: return chain.tip().difficulty recent = chain.get_block(height) old = chain.get_block(height - interval) if recent is None or old is None: return chain.tip().difficulty elapsed = recent.timestamp - old.timestamp expected = TARGET_BLOCK_TIME * interval ratio = elapsed / expected current_difficulty = recent.difficulty if ratio < 0.25: return current_difficulty + 2 elif ratio > 4.0: return max(1, current_difficulty - 2) elif ratio < 0.5: return current_difficulty + 1 elif ratio > 2.0: return max(1, current_difficulty - 1) return current_difficulty def fork_choice(chain_a, chain_b): """Return the chain with more accumulated work (heaviest chain rule).""" def total_work(chain): return sum(2 ** chain.get_block(i).difficulty for i in range(chain.height() + 1)) return chain_a if total_work(chain_a) >= total_work(chain_b) else chain_b

""" Two-way peg with SPV verification. Peg-in: Bitcoin → BTCv27 (mint wrapped BTC) Peg-out: BTCv27 → Bitcoin (burn wrapped BTC, release Bitcoin) Both directions are SPV-verified against the Bitcoin header chain. The TLA+ spec (tla/PegAtomicity.tla) formally models the state machine. """ from __future__ import annotations import hashlib import logging from typing import Optional from btcv27.core.exceptions import PegError from btcv27.core.transaction import Transaction logger = logging.getLogger(__name__) MIN_CONFIRMATIONS = 6 FEDERATION_ADDRESS = None # Loaded from env at runtime class TwoWayPeg: """ Manages peg-in and peg-out operations. State machine (see tla/PegAtomicity.tla for formal model): Peg-in: PENDING → CONFIRMED → MINTED Peg-out: PENDING → BURNED → RELEASED """ def __init__(self, bitcoin_rpc, spv_verifier, chain, storage): self._rpc = bitcoin_rpc self._spv = spv_verifier self._chain = chain self._storage = storage self._pending_pegins: dict = {} self._pending_pegouts: dict = {} async def process_peg_in( self, btc_tx_id: str, recipient_address: str, spv_proof: dict, ) -> Optional[Transaction]: """ Verify a Bitcoin peg-in transaction and mint wrapped BTC. 1. Verify the SPV proof against the Bitcoin header chain. 2. Check confirmation count. 3. Verify output is to the federation multisig address. 4. Check replay protection (tx not already processed). 5. Mint equivalent wrapped BTC on BTCv27. """ # Replay protection if btc_tx_id in self._pending_pegins: raise PegError(f"Peg-in {btc_tx_id} already processed.") # SPV proof verification if not self._spv.verify_transaction(btc_tx_id, spv_proof): raise PegError(f"SPV proof invalid for peg-in {btc_tx_id}.") # Confirmation check confirmations = await self._rpc.get_confirmations(btc_tx_id) if confirmations < MIN_CONFIRMATIONS: raise PegError( f"Peg-in {btc_tx_id} has {confirmations} confirmations " f"(need {MIN_CONFIRMATIONS})." ) # Verify amount from SPV proof (not from user input) amount = spv_proof.get("amount") if not amount or amount <= 0: raise PegError("Invalid peg-in amount in SPV proof.") # Mark as processed before minting (prevents TOCTOU race) self._pending_pegins[btc_tx_id] = True # Mint wrapped BTC mint_tx = Transaction( inputs=[{"tx_id": "peg_in", "output_index": -1}], outputs=[{ "recipient": recipient_address, "commitment": "0" * 64, "value": amount, }], fee=0.0, nonce=0, script={"type": "transfer", "params": {"peg_in_tx": btc_tx_id}}, ) logger.info( "Minted %.8f wBTC for peg-in %s → %s", amount, btc_tx_id, recipient_address ) return mint_tx async def process_peg_out( self, btcv27_tx: Transaction, btc_destination: str, ) -> str: """ Burn wrapped BTC on BTCv27 and release Bitcoin. 1. Verify the burn transaction on BTCv27. 2. Calculate peg-out amount from burned outputs. 3. Submit Bitcoin release transaction via federation. 4. Return the Bitcoin transaction ID. """ burn_amount = sum(o.value for o in btcv27_tx.outputs) if burn_amount <= 0: raise PegError("Peg-out burn amount must be positive.") # Submit Bitcoin release btc_tx_id = await self._rpc.send_to_address( btc_destination, burn_amount, fee_rate="economical" ) self._pending_pegouts[btcv27_tx.tx_id] = btc_tx_id logger.info( "Released %.8f BTC to %s (btc_tx=%s)", burn_amount, btc_destination, btc_tx_id ) return btc_tx_id

""" SPV (Simplified Payment Verification) proof verification. Enhancement 6: SPV verification now reads from the HeaderStore rather than requiring full block data, enabling light-client-grade verification. """ from __future__ import annotations import hashlib import logging from typing import Dict, List, Optional, Tuple logger = logging.getLogger(__name__) class SPVVerifier: """ Verifies Bitcoin transaction inclusion using Merkle proofs and the header chain. """ def __init__(self, header_store=None): self._header_store = header_store def verify_transaction( self, tx_id: str, proof: Dict, ) -> bool: """ Verify that a transaction is included in a Bitcoin block. proof must contain: - block_hash : Bitcoin block hash - merkle_proof : [(sibling_hash, 'left'|'right'), ...] - block_height : Block height """ try: block_hash = proof.get("block_hash") merkle_proof = proof.get("merkle_proof", []) block_height = proof.get("block_height", 0) # Verify Merkle proof leaf_hash = self._double_sha256(bytes.fromhex(tx_id)) if not self._verify_merkle_proof(leaf_hash, merkle_proof, block_hash): logger.warning("Merkle proof verification failed for tx %s", tx_id) return False # If we have a header store, verify the block hash is in our chain if self._header_store: header = self._header_store.get_by_hash(block_hash) if header is None: logger.warning( "Block hash %s not found in header store", block_hash[:12] ) return False return True except Exception as exc: logger.warning("SPV verification error: %s", exc) return False def _double_sha256(self, data: bytes) -> str: return hashlib.sha256(hashlib.sha256(data).digest()).hexdigest() def _verify_merkle_proof( self, leaf_hash: str, proof: List[Tuple[str, str]], expected_root: str, ) -> bool: current = leaf_hash for sibling, direction in proof: if direction == "left": combined = bytes.fromhex(sibling) + bytes.fromhex(current) else: combined = bytes.fromhex(current) + bytes.fromhex(sibling) current = self._double_sha256(combined) return current == expected_root

"""On-chain governance: proposal lifecycle and voting.""" from __future__ import annotations import time import uuid from dataclasses import dataclass, field from enum import Enum from typing import Dict, Optional from btcv27.core.exceptions import GovernanceError class ProposalStatus(str, Enum): ACTIVE = "active" APPROVED = "approved" REJECTED = "rejected" EXECUTED = "executed" @dataclass class Proposal: id: str title: str proposer: str created_at: float status: ProposalStatus = ProposalStatus.ACTIVE votes: Dict[str, bool] = field(default_factory=dict) vote_weights: Dict[str, float] = field(default_factory=dict) yes_weight: float = 0.0 no_weight: float = 0.0 class GovernanceEngine: """ Simple on-chain governance engine. Proposals are approved when yes_weight > 51% of total_supply and total votes >= 10% quorum. """ APPROVAL_THRESHOLD = 0.51 QUORUM_THRESHOLD = 0.10 def __init__(self, total_supply: float = 21_000_000.0): self._proposals: Dict[str, Proposal] = {} self._total_supply = total_supply def propose(self, title: str, proposer: str) -> str: pid = str(uuid.uuid4()) self._proposals[pid] = Proposal( id=pid, title=title, proposer=proposer, created_at=time.time(), ) return pid def vote( self, proposal_id: str, voter: str, approve: bool, weight: float = 1.0 ) -> None: proposal = self._proposals.get(proposal_id) if proposal is None: raise GovernanceError(f"Proposal {proposal_id} not found.") if proposal.status != ProposalStatus.ACTIVE: raise GovernanceError(f"Proposal {proposal_id} is not active.") if voter in proposal.votes: raise GovernanceError(f"Voter {voter} has already voted.") proposal.votes[voter] = approve proposal.vote_weights[voter] = weight if approve: proposal.yes_weight += weight else: proposal.no_weight += weight self._check_thresholds(proposal) def _check_thresholds(self, proposal: Proposal) -> None: total_voted = proposal.yes_weight + proposal.no_weight quorum_met = total_voted >= self._total_supply * self.QUORUM_THRESHOLD if not quorum_met: return yes_ratio = proposal.yes_weight / total_voted if yes_ratio > self.APPROVAL_THRESHOLD: proposal.status = ProposalStatus.APPROVED elif (1.0 - yes_ratio) > self.APPROVAL_THRESHOLD: proposal.status = ProposalStatus.REJECTED def get_proposal(self, proposal_id: str) -> Optional[Proposal]: return self._proposals.get(proposal_id) def list_active(self): return [p for p in self._proposals.values() if p.status == ProposalStatus.ACTIVE]
